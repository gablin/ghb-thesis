\input{figures/solving-techniques/interchangeable-data-example-styles}%
\begin{tikzpicture}
  \node [value node] (v) {};
  \node [computation node, position=-135 degrees from v] (cp1) {\opCopy};
  \node [computation node, position=- 45 degrees from v, dead] (cp2) {\opCopy};
  \node [value node, below=of cp1] (v1) {\opVar{v}[1]};
  \node [value node, below=of cp2, dead] (v2) {\opVar{v}[2]};
  \node [alt node, below=of v1] (alt1) {$\mVar{alt}[p_1]$};
  \node [alt node, below=of v2] (alt2) {$\mVar{alt}[p_2]$};

  \begin{scope}[data flow]
    \draw (v) -- (cp1);
    \draw (cp1) -- (v1);
    \draw (v1) -- (alt1);
    \draw (v1) -- (alt2.north west);

    \begin{scope}[dead]
      \draw (v) -- (cp2);
      \draw (cp2) -- (v2);
      \draw (v2) -- (alt2);
    \end{scope}
  \end{scope}

  \begin{pgfonlayer}{background}
    % m1
    \node [outer match node, inner sep=2pt, fit=(v)] (m1a) {};
    \node [outer match node, inner sep=2pt, fit=(v1)] (m1b) {};
    \node [outer match node, fit=(cp1)] (m1c) {};
    \def\pathMI{
      [bend left=45]
      (m1a.north west)
      to
      (m1a.north east)
      to
      (m1a.south east)
      [rounded corners=8pt]
      --
      (m1b.east |- m1c.east)
      [rounded corners=0]
      --
      (m1b.east)
      to
      (m1b.south)
      to
      (m1b.west)
      [rounded corners=8pt]
      --
      (m1b.west |- m1c.north)
      [rounded corners=0]
      --
      cycle
    }
    \draw [selected match line, fill=shade1] \pathMI;
  \end{pgfonlayer}
\end{tikzpicture}%
