\begingroup%
\figureFont\figureFontSize%
\pgfdeclarelayer{background}%
\pgfsetlayers{background,main}%
\begin{tikzpicture}[
    outer match node/.style={%
      match node,
      draw=none,
      inner sep=0,
    },
    selected/.style={
      solid,
    },
  ]

  \node [value node] (v) {\strut\opVar{v}};
  \node [computation node, position=-135 degrees from v] (cp1) {\opCopy};
  \node [computation node, position=- 45 degrees from v] (cp2) {\opCopy};
  \node [value node, below=of cp1] (v1) {\strut\opVar{v}[\hspace{-1pt}1]};
  \node [value node, below=of cp2] (v2) {\strut\opVar{v}[\hspace{-1pt}2]};
  \node [computation node, below=of v1] (op1) {};
  \node [computation node, below=of v2] (op2) {};

  \begin{scope}[data flow]
    \draw (v) -- (cp1);
    \draw (v) -- (cp2);
    \draw (cp1) -- (v1);
    \draw (cp2) -- (v2);
    \draw (v1) -- (op1);
    \draw (v2) -- (op2);
  \end{scope}

  \begin{pgfonlayer}{background}
    % m1
    \node [outer match node, inner sep=2.5pt, fit=(v1)] (m1a) {};
    \node [outer match node, fit=(op1)] (m1b) {};
    \def\pathMI{
      [bend left=45]
      (m1a.west)
      to
      (m1a.north)
      to
      (m1a.east)
      --
      (m1b.east)
      to
      (m1b.south)
      to
      (m1b.west)
      -- coordinate (m1)
      cycle
    }
    \path [fill=shade1]
          \pathMI;

    % m2
    \node [outer match node, inner sep=2.5pt, fit=(v2)] (m2a) {};
    \node [outer match node, fit=(op2)] (m2b) {};
    \def\pathMII{
      [bend left=45]
      (m2a.west)
      to
      (m2a.north)
      to
      (m2a.east)
      -- coordinate (m2)
      (m2b.east)
      to
      (m2b.south)
      to
      (m2b.west)
      --
      cycle
    }
    \path [fill=shade1]
          \pathMII;

    % m3
    \node [outer match node, inner sep=1.5pt, fit=(v)] (m3a) {};
    \node [outer match node, fit=(cp1)] (m3b) {};
    \node [outer match node, inner sep=1.5pt, fit=(v1)] (m3c) {};
    \def\pathMIII{
      [bend left=45]
      (m3a.north west)
      to
      (m3a.north east)
      to
      (m3a.south east)
      [rounded corners=4pt]
      --
      (m3b.east)
      [rounded corners=0]
      --
      (m3c.east)
      to
      (m3c.south)
      to
      (m3c.west)
      [rounded corners=4pt]
      --
      (m3c.west |- m3b.north west)
      [rounded corners=0]
      --
      cycle
    }
    \path [fill=shade1]
          \pathMIII;

    % m4
    \node [outer match node, inner sep=3pt, fit=(v)] (m4a) {};
    \node [outer match node, fit=(cp2)] (m4b) {};
    \node [outer match node, inner sep=1.5pt, fit=(v2)] (m4c) {};
    \def\pathMIV{
      [bend left=45]
      (m4a.south west)
      to
      (m4a.north west)
      to
      (m4a.north east)
      [rounded corners=4pt]
      --
      (m4c.east |- m4b.north east)
      [rounded corners=0]
      --
      (m4c.east)
      to
      (m4c.south)
      to
      (m4c.west)
      [rounded corners=4pt]
      --
      (m4b.west)
      [rounded corners=0]
      --
      cycle
    }
    \path [fill=shade1]
          \pathMIV;

    \begin{scope}
      \path [clip] \pathMI;
      \path [fill=shade2]
            \pathMIII;
    \end{scope}
    \begin{scope}
      \path [clip] \pathMII;
      \path [fill=shade2]
            \pathMIV;
    \end{scope}
    \begin{scope}
      \path [clip] \pathMIII;
      \path [fill=shade2]
            \pathMIV;
    \end{scope}

    \begin{scope}[selected]
      \draw \pathMI;
      \draw \pathMII;
      \draw \pathMIII;
      \draw \pathMIV;
    \end{scope}
  \end{pgfonlayer}

  % Match labels
  \node [match label, left=of m1] (m1l) {$\strut m_1$};
  \node [match label, right=of m2] (m2l) {$\strut m_2$};
  \foreach \i in {1, 2} {
    \draw [match attachment line] (m\i) -- (m\i l);
  }
\end{tikzpicture}%
\endgroup%
